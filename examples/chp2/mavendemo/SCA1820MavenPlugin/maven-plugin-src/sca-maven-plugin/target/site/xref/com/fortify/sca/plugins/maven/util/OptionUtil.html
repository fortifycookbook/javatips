9   public class OptionUtil {
10      public static void setSwitchOption(List<String> options, String option, boolean isRequired) {
11          if (isRequired) {
12              options.add(option);
13          }
14      }
15  
16      public static void setJvmOption(List<String> options, String option, String value) {
17          if (!StringUtil.isEmpty(value)) {
18              options.add(option + value);
19          }
20      }
21  
22      public static void setOption(List<String> options, String option, String value) {
23          if (!StringUtil.isEmpty(value)) {
24              options.add(option);
25              options.add(value);
26          }
27      }
28  
29      public static void setOption(List<String> options, String option, Integer value) {
30          if (value != null && value > 0) {
31              options.add(option);
32              options.add(String.valueOf(value));
33          }
34      }
35  
36      public static void setOption(List<String> options, String option, String[] values) {
37          if (values != null) {
38              for (String value : values) {
39                  options.add(option);
40                  options.add(value);
41              }
42          }
43      }
44  
45      public static void setProperty(List<String> options, String property, String value) {
46          if (!StringUtil.isEmpty(value)) {
47              options.add("-D" + property + "=" + value);
48          }
49      }
50  
51      public static void printOption(Log log, List<String> options, String option, String displayName)
52      {
53          if (!StringUtil.isEmpty(option)) {
54              int index = options.indexOf(option);
55              if (index != -1 && index < options.size() - 1) {
56                  log.info(displayName + ": " + options.get(index + 1));
57              }
58          }
59      }
